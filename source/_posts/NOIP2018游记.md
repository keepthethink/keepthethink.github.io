	如果你问"isprint()函数是什么"或"isalnum()函数是什么",请自行忽略这段话并自觉编写判断函数。

代码：
```cpp
#include<iostream>
#include<string>
using namespace std;

bool check(char c) {
    if('0' <= c && c <= '9') {
        return true;
    }
    if('a' <= c && c <= 'z') {
        return true;
    }
    if('A' <= c && c <= 'Z') {
        return true;
    }
    return false;
}

int main() {
    string s;
    getline(cin, s);
    
    int len = s.length();
    
    int ans = 0;
    for(int i = 0; i < len; i++) {
        if(check(s[i])) {
            ans++;
        }
    }
    cout << ans << endl;
    
    return 0;
}
```

2. 用一个光标变量记录目前接到了第几个学生，如果还有人没被接到，就继续循环知道所有人都被接到为止。
